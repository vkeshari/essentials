### git ###

git clone <url>

git status

# Sync workspace to remote
git pull  # if no unchanged files
git merge # if changed files

# Staging and unstaging
git add <filename>
git rm <filename>
git restore --staged <filename>

# Committing staged changes
git commit -m '<message>'
git push origin master

git checkout <branch>

# Reverting a file
git restore <filename>
git restore --source <commithash> <filename>
git restore --source <branch> <filename>

# Reverting workspace
git restore .
git reset <commithash> # verify this

# Checkout Branch
git branch
git checkout master
git checkout <branch>

# Create Branch (alternatives)
git branch <newbranch>
git checkout -b <branch>
git checkout -b <branch> <frombranch>



### pip ###

pip check
pip list

# Upgrade pip
python -m pip install --upgrade pip

# Install packages
pip install <pkg>
pip uninstall <pkg>

# Upgrade packages
pip -U install <pkg> # --upgrade


### brew ###

brew update
brew upgrade

brew list

brew install <pkg>
